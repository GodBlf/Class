tmpl := "你是{{.name}}"
	parse, err := template.New("my_template").Parse(tmpl)
	if err != nil {
		panic(err)
	}
	//buffer字节缓冲区相当于strings.builder容器
	bf := &bytes.Buffer{}
	parse.Execute(bf, map[string]any{
		"name": "大学老师",
	})
	fmt.Println(bf.String())
//字符串先解析(离散化)成模板,然后再再buffer中根据map拼接字符串存到buffer里
