```c
#include<iostream>
#include<vector>
#include<stack>

using namespace std;


int main() {
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	//
	int n; int m; int k;
	cin >> n; cin >> m; cin >> k;
	for (int kk = 0; kk < k; kk++) {
		//
		stack<int> huojia;
		huojia.push(10000);
		int need = 1;
		int cap = 1;
		int flag = 1;
		vector<int> arr(n, 0);
		for (int i = 0; i < n; i++) {
			cin >> arr[i];
		}
		for (int i = 0; i < n; i++) {
			while (true) {
				if (huojia.top() != need) {
					break;
				}
				huojia.pop();
				need++;
				cap--;
			}
			int num = arr[i];
			if (num == need) {
				need++;
				continue;
			}
			else {
				if (cap > m) {
					flag = 0;
					goto loop;
				}
				else {
					cap++;
					huojia.push(num);
				}
			}
		}
		while (true) {
			if (huojia.top() != need) {
				break;
			}
			huojia.pop();
			need++;
			cap--;
		}
		if (need != n + 1) {
			flag = 0;
		}
	loop:
		if (flag == 0) {
			cout << " no";
		}
		else {
			cout << "yes";
		}
	}
	//
}
```
